%% itembf macro family for formating item titles.
%% It takes as argument every token from \itembf until it finds a ':' token
\def\itembf#1:{
	\item\textbf{#1:}
}
\def\subitembf#1:{
	\subitem\textbf{#1:}
}
\def\subsubitembf#1:{
	\subitem\textbf{#1:}
}
