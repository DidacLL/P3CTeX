%% CLASS REGISTRY
%% Global registry for all UML classes managed by pxUML.
%% Each logical class name is mapped (via pxPRP) to a backing prop
%% storing attributes, operations and other metadata.
\prop_if_exist:NF \g_pxuml_classREG_prop {\prop_gclear_new:N \g_pxuml_classREG_prop}
%% =================================
%% ==== pxUML KEYS DEFINITION ===
%% =================================
\keys_define:nn { pxUML }
{
	maincolor	 	.tl_set:N		=	\l__pxuml_main_color_tl,
		maincolor		.initial:n	=	UMLprimaryColor,
	auxcolor	 	.tl_set:N		=	\l__pxuml_aux_color_tl,
		auxcolor		.initial:n	=	UMLsecondaryColor,
	neutralcolor	.tl_set:N		=	\l__pxuml_neutral_color_tl,
		neutralcolor	.initial:n	=	UMLneutralColor,
	bgcolor 		.tl_set:N		=	\l__pxuml_bg_color_tl,
		bgcolor			.initial:n	=	UMLBackgroundColor,
}
\ProcessKeysOptions{ pxUML }
%% PCKG PRIVATE FUNCTIONS
%% NEW CLASS
%%	#1 - CLASS NAME
%%	#2 - ATTRIBUTES
%%	#3 - OPERATIONS
%%	#4 - INHERITANCE
%%	#5 - OBJ-TYPE
%%	#6 - VISIBILITY
%%	#7 - IMPLEMENTATION
\cs_new:Npn \__pxuml_newclass:nnnnnnn #1#2#3#4#5#6#7{
	\pxPRP_set:nnnN	{#1}{ attributes }	{ #2 } 	\g_pxuml_classREG_prop
	\pxPRP_set:nnnN	{#1}{ operations }	{ #3 } 	\g_pxuml_classREG_prop
	\pxPRP_set:nnnN	{#1}{ inheritance}	{ #4 } 	\g_pxuml_classREG_prop
	\pxPRP_set:nnnN	{#1}{ objtype 	 }	{ #5 } 	\g_pxuml_classREG_prop
	\pxPRP_set:nnnN	{#1}{ visibility }	{ #6 } 	\g_pxuml_classREG_prop
	\pxPRP_set:nnnN	{#1}{ implement  }	{ #7 } 	\g_pxuml_classREG_prop
}
%% NEW CLASS VARIANT
\cs_generate_variant:Nn \__pxuml_newclass:nnnnnnn	{eeeeeee}
\cs_new:Npn \__pxuml_newclass:nnn #1#2#3{\__pxuml_newclass:nnnnnnn {#1}{#2}{#3}{}{class}{public}{}}
%% ADD ATTRIBUTE
%%	#1 - CLASS NAME
%%	#2 - ATTRIBUTE
\cs_new:Npn \__pxuml_add_attribute:nn #1#2 {
	\pxPRP_put_items:nnnN {	#1 }{ attributes } {#2} \g_pxuml_classREG_prop
}
%% ADD INHERITANCE
%%	#1 - CLASS NAME
%%	#2 - INHERITANCE
\cs_new:Npn \__pxuml_add_inherit:nn #1#2 {
	\pxPRP_put_items:nnnN {	#1 }{ inheritance } {#2} \g_pxuml_classREG_prop
}
%% ADD OPERATION
%%	#1 - CLASS NAME
%%	#2 - OPERATION
\cs_new:Npn \__pxuml_add_operation:nn #1#2 {
	\pxPRP_put_items:nnnN {	#1 }{ operations } {#2} \g_pxuml_classREG_prop
}
%% DRAW CLASS
%%	#1 - CLASS NAME
%%	#2 - ATTRIBUTES
%%	#3 - OPERATIONS
\cs_new:Npn \__pxuml_drawclass:nnn #1#2#3 {
	%%%%%%%% ATTRIBUTES
	\clist_clear_new:N	\l__attributes_clist
	\tl_clear_new:N \l__attributes_tl
	\pxprp_get_clist:nnNN {#1}{ attributes } \l__attributes_clist \g_pxuml_classREG_prop
	%%%%%%%% OPERATIONS
	\clist_clear_new:N	\l__operations_clist
	\tl_clear_new:N \l__operations_tl
	\pxprp_get_clist:nnNN { #1 }{ operations } \l__operations_clist \g_pxuml_classREG_prop
	%%%%%%%% INHERITANCE
	\tl_clear_new:N \l__inheritance_tl
	\pxprp_get_tl:nnNN { #1 }{ inheritance } \l__inheritance_tl \g_pxuml_classREG_prop
	%%%%%%%% OBJ-TYPE
	\tl_clear_new:N \l__obj_type_tl
	\pxprp_get_tl:nnNN { #1 }{ objtype } \l__obj_type_tl \g_pxuml_classREG_prop
	%%%%%%%% visibilityy
	\tl_clear_new:N \l__visibility_tl
	\pxprp_get_tl:nnNN { #1 }{ visibility } \l__visibility_tl \g_pxuml_classREG_prop
	%%%%%%%% implementation
	\tl_clear_new:N \l__implementation_tl
	\pxprp_get_tl:nnNN { #1 }{ implement } \l__implementation_tl \g_pxuml_classREG_prop
	%%%%%%%% DRAW
	\begin{\l__obj_type_tl}[ultra~thick, #3]{#1}{#2}
		\tl_if_blank:VF \l__implementation_tl {\cs:w implement \cs_end:{\l__implementation_tl}}
		\clist_if_empty:NF \l__operations_clist {
			\clist_if_in:NVF \l__operations_clist \c_novalue_tl {
				\clist_map_inline:Nn \l__operations_clist {\cs:w operation \cs_end: {##1}}
			}
		}
		\clist_if_empty:NF \l__attributes_clist {
			\clist_if_in:NVF \l__attributes_clist \c_novalue_tl {
				\clist_map_inline:Nn \l__attributes_clist {\cs:w attribute \cs_end: {##1}}
			}
		}
		\tl_if_blank:VF \l__inheritance_tl {\cs:w inherit \cs_end:{\l__inheritance_tl}}
	\end{\l__obj_type_tl}
}
%% DRAW SIMPLE CLASS
%%	#1 - CLASS NAME
%%	#2 - ATTRIBUTES
%%	#3 - OPERATIONS
\cs_new:Npn \__pxuml_drawsimpleclass:nnn #1#2#3 {
	%%%%%%%% ATTRIBUTES
	\clist_clear_new:N	\l__attributes_clist
	\tl_clear_new:N \l__attributes_tl
	\pxprp_get_clist:nnNN {#1}{ attributes } \l__attributes_clist \g_pxuml_classREG_prop
	%%%%%%%% INHERITANCE
	\tl_clear_new:N \l__inheritance_tl
	\pxprp_get_tl:nnNN { #1 }{ inheritance } \l__inheritance_tl \g_pxuml_classREG_prop
	%%%%%%%% OBJ-TYPE
	\tl_clear_new:N \l__obj_type_tl
	\pxprp_get_tl:nnNN { #1 }{ objtype } \l__obj_type_tl \g_pxuml_classREG_prop
	%%%%%%%% visibilityy
	\tl_clear_new:N \l__visibility_tl
	\pxprp_get_tl:nnNN { #1 }{ visibility } \l__visibility_tl \g_pxuml_classREG_prop
	%%%%%%%% implementation
	\tl_clear_new:N \l__implementation_tl
	\pxprp_get_tl:nnNN { #1 }{ implement } \l__implementation_tl \g_pxuml_classREG_prop
	%%%%%%%% DRAW
	\begin{\l__obj_type_tl}[ultra~thick, #3]{#1}{#2}
		\foreach \objattr in \l__attributes_clist {\attribute{\objattr}}
		\tl_if_blank:VF \l__inheritance_tl {\cs:w inherit \cs_end:{\l__inheritance_tl}}
	\end{\l__obj_type_tl}
}
%% LOAD VARIABLE COLORS
\cs_new:Nn \__pxuml_load_varcolor:{
	\renewcommand {\umltextcolor }{\l__pxuml_bg_color_tl!}
	\renewcommand {\umlfillcolor }{\l__pxuml_main_color_tl!40!\l__pxuml_neutral_color_tl!60!}
	\renewcommand {\umldrawcolor }{\l__pxuml_bg_color_tl!}
}
%% LOAD MAIN COLORS
\cs_new:Nn \__pxuml_load_maincolor:{
	\renewcommand {\umltextcolor }{\l__pxuml_main_color_tl}
	\renewcommand {\umlfillcolor }{\l__pxuml_aux_color_tl!20!}
	\renewcommand {\umldrawcolor }{\l__pxuml_main_color_tl}
}
%% SHOW CLASS (INTERNAL DATA LOAD)
%%	#1 - CLASS NAME
\cs_new:Npn \__pxuml_showclass:n #1 {
	\pxprp_get_tl:nnNN { #1 }{ attributes } \l__attributes_tl \g_pxuml_classREG_prop
	\pxprp_get_tl:nnNN { #1 }{ operations } \l__operations_tl \g_pxuml_classREG_prop
	\pxprp_get_tl:nnNN { #1 }{ inheritance } \l__inheritance_tl \g_pxuml_classREG_prop
	\pxprp_get_tl:nnNN { #1 }{ objtype } \l__obj_type_tl \g_pxuml_classREG_prop
	\pxprp_get_tl:nnNN { #1 }{ visibility } \l__visibility_tl \g_pxuml_classREG_prop
	\pxprp_get_tl:nnNN { #1 }{ implement } \l__implementation_tl \g_pxuml_classREG_prop
}

%% INSPECTION HELPERS (READ-ONLY)
%%	#1 - CLASS NAME
%% Expand to stored attributes / operations / inheritance respectively.
\cs_new:Npn \__pxuml_print_attributes:n #1 {
	\tl_clear_new:N \l__pxuml_attrs_out_tl
	\pxprp_get_tl:nnNN { #1 }{ attributes } \l__pxuml_attrs_out_tl \g_pxuml_classREG_prop
	\tl_use:N \l__pxuml_attrs_out_tl
}
\cs_new:Npn \__pxuml_print_operations:n #1 {
	\tl_clear_new:N \l__pxuml_ops_out_tl
	\pxprp_get_tl:nnNN { #1 }{ operations } \l__pxuml_ops_out_tl \g_pxuml_classREG_prop
	\tl_use:N \l__pxuml_ops_out_tl
}
\cs_new:Npn \__pxuml_print_parents:n #1 {
	\tl_clear_new:N \l__pxuml_inh_out_tl
	\pxprp_get_tl:nnNN { #1 }{ inheritance } \l__pxuml_inh_out_tl \g_pxuml_classREG_prop
	\tl_use:N \l__pxuml_inh_out_tl
}
