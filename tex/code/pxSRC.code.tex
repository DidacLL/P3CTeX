\cs_new_protected:Npn \__pxsrc_label_from_string:nN #1 #2
  {
    \tl_set:Nn #2 { #1 }
    % Drop a leading ./ which is common in relative paths.
    \regex_replace_once:nnN { \A \./ } { } #2
    % Keep only characters that are safe for \label (csname-based storage).
    \regex_replace_all:nnN { [^-A-Za-z0-9:_.]+ } { - } #2
    \regex_replace_all:nnN { -+ } { - } #2
    \tl_set:Nx #2 { \l__pxsrc_label_prefix_tl \tl_use:N #2 }
  }

\cs_new:Npn \__pxsrc_resolve_graphic:nN #1 #2
  {
    \tl_clear:N #2
    \file_if_exist:nTF { #1 }
      { \tl_set:Nn #2 { #1 } }
      {
        \file_if_exist:nTF { #1 .png }
          { \tl_set:Nn #2 { #1 .png } }
          {
            \file_if_exist:nTF { #1 .pdf }
              { \tl_set:Nn #2 { #1 .pdf } }
              {
                \file_if_exist:nTF { #1 .jpg }
                  { \tl_set:Nn #2 { #1 .jpg } }
                  { }
              }
          }
      }
  }

\cs_new:Npn \__pxsrc_graphic_box:nnn #1 #2 #3
  {
    \tl_clear:N \l__pxsrc_graphic_file_tl
    \__pxsrc_resolve_graphic:nN { #2 } \l__pxsrc_graphic_file_tl
    \tl_if_blank:VTF \l__pxsrc_graphic_file_tl
      { \__px_fancy_shadowbox:nnnnn { #1 } { ``~ #2 ~'' } { ~~>>ERR! #2.png} {IMAGE~ NOT~FOUND } { #3 } }
      { \includegraphics[width = #1]{ \l__pxsrc_graphic_file_tl } }
  }
