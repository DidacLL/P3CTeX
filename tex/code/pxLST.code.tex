      \def\pxlst@colback@v  { darcula-bg!90! }

      \def\pxlst@colframe@v { gray!15! }
      \def\pxlst@coltext@v  { black }
      \def\pxlst@lstyle@v   { pxlst@plain }

  }
  {
    \def\pxlst@colback@v  { darcula-bg!90! }
    \def\pxlst@colframe@v { darcula-bg!60! }
    \def\pxlst@coltext@v  { darcula-fg }

  \edef\pxlst@numbers@v { \tl_use:N \l_pxlst_numbers_tl  }

    \noexpand\ttfamily
  }
}

%% ===== PGF pxlst@configure KEY ====
%% =================================
\tcbset{
  pxlst@configure/.code={
    \tl_if_blank:nF {#1} { \pxLST_set_options:n {#1} }
    \tcbset{
      colback  = \pxlst@colback@v,
      colframe = \pxlst@colframe@v,
      coltext  = \pxlst@coltext@v,
    }
    \__pxlst_apply_lstset:
  },
}
%% \__pxlst_apply_lstset: — build a fully-expanded \lstset call and execute it.
%% The \edef approach expands display macros (\pxlst@lstyle@v → pxlst@dark, etc.)
%% into literal strings so that listings can find the defined style/language names.
\cs_new_protected:Npn \__pxlst_apply_lstset: {
  \edef \pxlst@callbuf {
    \noexpand\lstset{
      style=\pxlst@lstyle@v,
      language=\pxlst@lang@v,
      numbers=\pxlst@numbers@v,
      tabsize=\pxlst@tabsize@v
    }
  }
  \pxlst@callbuf
  \lstset{basicstyle=\pxlst@basicstyle@v}
}

%% ===== PROBE DIAGNOSTIC ===========
%% =================================
\cs_new_protected:Npn \__pxlst_probe_emit: {
  \bool_if:NT \l_pxlst_probe_bool
  {
    \iow_term:x { PXLST_ASSERT:STYLE=\tl_use:N \l_pxlst_style_tl }
    \iow_term:x { PXLST_ASSERT:LANGUAGE=\tl_use:N \l_pxlst_language_tl }
    \iow_term:x { PXLST_ASSERT:NUMBERS=\tl_use:N \l_pxlst_numbers_tl }
    \iow_term:x { PXLST_ASSERT:COLBACK=\pxlst@colback@v }
    \iow_term:x { PXLST_ASSERT:LSTYLE=\pxlst@lstyle@v }
  }
}

%% ===== PRESET FUNCTIONS ===========

\cs_new_protected:Npn \pxLST_save_preset:nn #1 #2 {
  \prop_put:Nnn \l_pxlst_presets_prop { #1 } { #2 }
}

\cs_new_protected:Npn \pxLST_use_preset:n #1 {
  \prop_get:NnNTF \l_pxlst_presets_prop { #1 } \l__pxlst_retrieved_tl
    { \exp_args:NV \pxLST_set_options:n \l__pxlst_retrieved_tl }
    { \msg_warning:nnn { pxLST } { unknown-preset } { #1 } }
}

%% ===== FILE INPUT FUNCTION ========
%% =================================
\cs_new_protected:Npn \pxLST_input_file:nn #1 #2 {
  \group_begin:
    \tl_if_blank:nF {#1} { \pxLST_set_options:n {#1} }
    \__pxlst_set_style_vars:
    \tcbinputlisting{
      pxlst@configure={},
      listing~file=#2,
      listing~only,
      arc=.25em, outer~arc=.25em,
      drop~lifted~shadow,
      toptitle=-.1em,
      title=\small\hspace*{\fill}\textbf{\ttfamily #2}
    }
  \group_end:
}
