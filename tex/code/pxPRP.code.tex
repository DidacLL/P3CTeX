%% MAIN VARIABLES
%% PROP NAME REG register all created pxPROPS	INTERNAL
\prop_if_exist:NF \g_pxprp__pxPRP_propREG_prop
{
	\prop_gclear_new:N \g_pxprp__pxPRP_propREG_prop
}
\cs_new:Npn \pxPRP_REG:n #1 {
	\prop_if_exist:cF {g_pxprp__pxPRP_ #1 REG_prop} {\prop_new:c {g_pxprp__pxPRP_ #1 REG_prop}}
	\use:c {g_pxprp__pxPRP_ #1 REG_prop}
}
%% GET PROP NAME
%%  #1 - PROP NAME
%%  #2 - PROP REGISTER (INTERNAL)
%%  #3 - RETURN VALUE tl var
\cs_new:Npn \pxPRP_get_propname:nNN #1 #2 #3{
	\prop_get:NnN #2 { #1 } #3
}
\cs_new:Npn \pxPRP_get_propname:nN #1 #2{
	\prop_get:NnN \g_pxprp__pxPRP_propREG_prop {#1} #2
}
\prg_new_conditional:Npnn \pxPRP_get_propname:nNN #1 #2 #3 { T , F, TF }{
	\prop_get:NnNTF #2 { #1 } #3 {	\prg_return_true:}{ \prg_return_false:}
}
\prg_new_conditional:Npnn \pxPRP_get_propname:nN #1 #2 { T , F, TF }{
	\prop_get:NnNTF \g_pxprp__pxPRP_propREG_prop {#1} #2 {	\prg_return_true:}{	\prg_return_false:	}
}
%% BUILD PROP NAME
%%  #1 - PROP NAME
%%  #2 - tlvar
\cs_new:Npn \__pxprp_build_name:nN #1 #2{
	\str_set:Ne #2 {#1}
	\str_put_right:Ne #2 {_prop}
	\str_put_left:Ne #2 {l_pxprp_}
}
\cs_new:Npn \__pxprp_build_name:n #1{
	\tl_clear_new:N \l_pxprp_name_tmp_tl
	\__pxprp_build_name:nN {#1} \l_pxprp_name_tmp_tl
	\tl_use:N  \l_pxprp_name_tmp_tl
}
%% NEW PROP
%% Creates new PROP object and registers its existence into REGISTER (INTERNAL)
%%  #1 - PROP NAME
%%  #2 - PROP REGISTER (INTERNAL)
\cs_new:Npn \pxPRP_new_prop:nN #1 #2 {
	\tl_clear_new:N  \l_pxprp_regnew_tmpa_tl
	\__pxprp_build_name:nN {	#1	}		\l_pxprp_regnew_tmpa_tl
	\prop_if_exist:cF		{ 	\tl_use:N	\l_pxprp_regnew_tmpa_tl	}
	{
		\prop_new:c 		{	\tl_use:N 	\l_pxprp_regnew_tmpa_tl	}
		\prop_gput:NnV	#2	{	#1 	}		\l_pxprp_regnew_tmpa_tl
	}
}
%% ENSURE PROP EXISTS AND CREATE IT IF IT DOES NOT EXIST
%%  #1 - PROP NAME
%%  #2 - PROP REGISTER (INTERNAL)
\cs_new:Npn \pxPRP_ensure:nN #1 #2 {
	\prop_if_in:NnF		#2 		{	#1	}	{
		\pxPRP_new_prop:nN		{	#1	}	#2
	}
}
\cs_new:Nn \pxPRP_ensure:n {
	\prop_if_exist:NF \g_pxprp__pxPRP_propREG_prop {
		\prop_gclear_new:N \g_pxprp__pxPRP_propREG_prop
	}
	\pxPRP_ensure:nN {#1} \g_pxprp__pxPRP_propREG_prop
}
%% GET PROP
%% Get the full qualified name for PROP from prop REGISTER (INTERNAL)
%%  #1 - PROP NAME
%%  #2 - PROP REGISTER (INTERNAL)
%%  #3 - RETURN VALUE tl var
\cs_new:Npn \pxPRP_getprop:nNN #1 #2 #3{
	\pxPRP_ensure:nN { #1 } #2
	\prop_get:NnN #2 {#1} #3
}
\cs_new:Npn \pxPRP_getprop:nN #1 #2{
	\pxPRP_ensure:n { #1 }
	\prop_get:NnN #2 {#1} \g_pxprp__pxPRP_propREG_prop
%	 \pxPRP_get_propname:nN {#1} #2
}
%%%%	PROP KEYS MANAGER >>>>
%% SET NEW KEY
%%  #1 - PROP ID NAME
%%  #2 - KEY
%%  #3 - VALUE
%%  #4 - PROP REGISTER (INTERNAL)
\cs_new:Npn \pxPRP_set:nnnN #1 #2 #3 #4{
	\tl_clear_new:N \l_pxprp_set_propfullname_tl
	\pxPRP_getprop:nNN {#1} #4 \l_pxprp_set_propfullname_tl
	\prop_if_in:cnTF {\l_pxprp_set_propfullname_tl} { #2 }
	{
		\prop_remove:cn {\l_pxprp_set_propfullname_tl} { #2 }
	}{
	}
	\prop_put:cnn {\l_pxprp_set_propfullname_tl} {#2} {#3}
}
%% SET NEW KEY
%%  #1 - PROP NAME
%%  #2 - KEY
%%  #3 - VALUE
%%  #4 - PROP REGISTER (INTERNAL)
\cs_new:Npn \pxPRP_set:nnn #1 #2 #3{
	\tl_clear_new:N \l_pxprp_set_propfullname_tl
	\pxPRP_getprop:nN {#1} \l_pxprp_set_propfullname_tl
	\prop_if_in:cnTF {\l_pxprp_set_propfullname_tl} { #2 }
	{
		\prop_remove:cn {\l_pxprp_set_propfullname_tl} { #2 }
	}{
	}
	\prop_put:cnn {\l_pxprp_set_propfullname_tl} {#2} {#3}
}
%% PUT ITEMS IN A LIST
%% Set value for a key in a PROP object as a list
%%  #1 - PROP ID NAME
%%  #2 - KEY
%%  #3 - VALUE
%%  #4 - PROP REGISTER (INTERNAL)
\cs_new:Npn \pxPRP_put_items:nnnN #1 #2 #3 #4{
	\tl_clear_new:N \l_pxprp_put_item_propfullname_tl
	\pxPRP_getprop:nNN {#1} #4 \l_pxprp_put_item_propfullname_tl
	\clist_clear_new:N \l_pxprp_put_item_aux_clist
	\prop_get:cnNTF {\l_pxprp_put_item_propfullname_tl} { #2 } \l_pxprp_put_item_aux_clist
	{	\clist_put_right:Nn \l_pxprp_put_item_aux_clist {	#3	}	}
	{	\clist_set:Nn \l_pxprp_put_item_aux_clist {	#3	}  	}
	\prop_put:cnV {\l_pxprp_put_item_propfullname_tl} {#2} \l_pxprp_put_item_aux_clist
}
%% PUT ITEMS IN A LIST
%% Set value for a key in a PROP object as a list
%%  #1 - PROP NAME
%%  #2 - KEY
%%  #3 - VALUE
%%  #4 - PROP REGISTER (INTERNAL)
\cs_new:Npn \pxPRP_put_items:nnn #1 #2 #3{
	\tl_clear_new:N \l_pxprp_put_item_propfullname_tl
	\clist_clear_new:N \l_pxprp_put_item_aux_clist

	\pxPRP_getprop:nN {#1} \l_pxprp_put_item_propfullname_tl

	\prop_get:cnNTF {\l_pxprp_put_item_propfullname_tl} { #2 } \l_pxprp_put_item_aux_clist {
		\clist_put_right:Nn 	\l_pxprp_put_item_aux_clist 	{	#3	}
	}{
		\clist_set:Nn 	\l_pxprp_put_item_aux_clist 	{	#3	}
	}
	\prop_put:cnV {	\l_pxprp_put_item_propfullname_tl	} {	#2	} 	\l_pxprp_put_item_aux_clist
}

%% CLEAR KEY
%%  #1 - PROP NAME
%%  #2 - KEY
%%  #3 - PROP REGISTER (INTERNAL)
\cs_new:Npn \__pxprp_clear_key:nnN  #1 #2 #3{
	\tl_clear_new:N 		\l_strlst_clearlist_name_tl
	\pxPRP_get_propname:nNNT {#1}  #3 \l_strlst_clearlist_name_tl {
		\prop_if_in:cnT {\l_strlst_clearlist_name_tl} {#2} {
			\prop_remove:cn	{#1	} {	#2	}
		}
	}
}
%%%%   GET PROP ITEM

\cs_new:Npn \pxprp_get_clist:nnNN #1 #2 #3 #4 {

	 {	\prop_get:cnN {\l_pxprp_gtlst_name_tl} {#2} #3	}
	 {	\clist_clear_new:N #3	}
}
%% GET ITEM AS A TL

%%  #1 - PROP NAME
%%  #2 - KEY
%%  #3 - RETURN VALUE TL

\cs_new:Npn \pxprp_get_tl:nnNN #1 #2 #3 #4 {

	 \pxPRP_get_propname:nNNTF 	{	#1	}	#4   \l_pxprp_gettl_name_tl

\cs_new:Npn \pxprp_get_clist:nnN #1 #2 #3 {	
	\pxprp_get_clist:nnNN {#1}{#2} #3 \g_pxprp__pxPRP_propREG_prop
}
%% 	GET CLIST ITEM VALUE
%%  #1 - PRPLST NAME tl
%%  #2 - PRPLST KEY tl
%%  #3 - CLIST INT POSITION
%%  #4 - retVal clist
%%  #5 - Registry prop (Opt.)
\cs_new:Npn \pxPRP_get_item:nnnNN #1 #2 #3 #4 #5 {
	\clist_clear_new:N \l_pxprp_get_item_aux_clist
	\pxprp_get_clist:nnNN{#1}{#2} \l_pxprp_get_item_aux_clist #5
	\tl_set:Nn #4 { \clist_item:Nn \l_pxprp_get_item_aux_clist { #3 }}
}
\cs_new:Npn \pxPRP_get_item:nnnN #1 #2 #3 #4 {
	\clist_clear_new:N \l_pxprp_get_item_aux_clist
	\pxprp_get_clist:nnN{#1}{#2} \l_pxprp_get_item_aux_clist
	\tl_set:Nn #4 { \clist_item:Nn \l_pxprp_get_item_aux_clist { #3 }}
}
\cs_new:Npn \pxPRP_get_item:nnn #1 #2 #3 {
	\clist_clear_new:N \l_pxprp_get_item_aux_clist
	\pxprp_get_clist:nnN{#1}{#2} \l_pxprp_get_item_aux_clist
	\clist_item:Nn \l_pxprp_get_item_aux_clist { #3 }
}

	\pxprp_get_clist:nnN{#1}{#2} \l_pxprp_get_item_aux_clist

}

%% 	GET as BOOLEAN
%%  #1 - PROP NAME
%%  #2 - PRPLST KEY tl
%%  #3 - retVal bool
%%  #4 - Registry prop (Opt.)
\cs_new:Npn \pxprp_get_bool:nnNN #1 #2 #3 #4 {
	\tl_clear_new:N \l_pxprp_getbool_name_tl
	\tl_clear_new:N \l_pxprp_gtbool_aux_tl
	 \pxPRP_get_propname:nNNTF {#1} #4 \l_pxprp_getbool_name_tl {
		\prop_get:cnN {\tl_use:N \l_pxprp_getbool_name_tl} {#2} \l_pxprp_gtbool_aux_tl
		\tl_if_blank:NTF \l_pxprp_gtbool_aux_tl {
			\bool_set_false:N  #3
		}{
			\tl_if_eq:NnTF \l_pxprp_gtbool_aux_tl {false} {
				\bool_set_false:N #3
			}{
				\bool_set_true:N #3
			}
		}
	}{
		\bool_set_false:N  #3
	}

}%%%%%% B O O L E A N - S H O R T    %%%%%
\cs_new:Npn \pxprp_get_bool:nnN #1 #2 #3 {
	\pxprp_get_bool:nnNN {#1}{#2} #3 \g_pxprp__pxPRP_propREG_prop
}
%%  PUT INTO LIST
%%  #1 - PROP NAME
%%  #2 - KEY
%%  #3 - COMMA SEPARATED LIST
%%  #4 - Registry prop (Opt.)
%%    R I G H T
\cs_new:Npn \pxprp_list_put_right:nnnN #1 #2 #3 #4 {
	%% ENSURE THAT PRPLIST EXIST OR CREATE IT
	\pxPRP_ensure:nN	{	#1	}	#4
	%% CREATE CLEAN VARIABLES
	\tl_clear_new:N 	\l_strlstputr_name_tl
	\tl_clear_new:N 	\l_strlstputr_rawlist_tl
	\clist_clear_new:N 	\l_strlstputr_list_clist
	\pxPRP_get_propname:nNNT { #1 } #4  \l_strlstputr_name_tl
	{
		\pxprp_get_clist:nnNN 	{	#1	}	{	#2	}	\l_strlstputr_list_clist	#4
		\clist_if_empty:NTF \l_strlstputr_list_clist
		{
			\prop_put:cnn	{	\l_strlstputr_name_tl	}	{	#2	}	{	#3	}
		}{
			\clist_put_right:Nn \l_strlstputr_list_clist {	#3	}
		}

		\prop_put:cnn	{	\l_strlstputr_name_tl	}	{	#2	}
		{
			\clist_use:Nn	\l_strlstputr_list_clist		{,}
		}
	}
}%%%%%%    R I G H T - S H O R T     %%%%%
\cs_new:Npn \pxprp_list_put_right:nnn #1 #2 #3 {
	 \pxprp_list_put_right:nnnN		{	#1	}	{	#2	}	{	#3	}	\g_pxprp__pxPRP_propREG_prop
}%%%%%%    L E F T                   %%%%%
\cs_new:Npn \pxprp_list_put_left:nnnN #1 #2 #3 #4 {
	\pxPRP_ensure:nN {	#1	}	#4
	\tl_clear_new:N 	\l_strlstputl_name_tl
	\tl_clear_new:N 	\l_strlstputl_rawlist_tl
	\clist_clear_new:N 	\l_strlstputl_list_clist
	%% OBTAIN PRPLST NAME
	\prop_get:NnN 	#4 	{	#1	}	\l_strlstputl_name_tl
	\prop_get:cnNTF {\tl_use:N \l_strlstputl_name_tl} {#2} \l_strlstputl_rawlist_tl
	{ %% OBTAIN PRPLST #1 >> KEY #2 >> LIST#1
		\clist_set:NV \l_strlstputl_list_clist \l_strlstputl_rawlist_tl
		\clist_put_left:Nn \l_strlstputl_list_clist {	#3	}
		\prop_put:cnn	{	\l_strlstputl_name_tl	}	{	#2	}
		{
			\clist_use:Nn	\l_strlstputl_list_clist		{,}
		}
	}{ %% PUT VALUE IN PRPLST #1 >> KEY #2 >> LIST#1
		\prop_put:cnn	{	\l_strlstputl_name_tl	}	{	#2	}	{	#3	}
	}
}%%%%%%    L E F T  - S H O R T      %%%%%
\cs_new:Npn \pxprp_list_put_left:nnn #1 #2 #3 {
	 \pxprp_list_put_left:nnnN		{	#1	}	{	#2	}	{	#3	}	\g_pxprp__pxPRP_propREG_prop
}
