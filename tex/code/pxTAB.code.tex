\cs_generate_variant:Nn \seq_set_split:Nnn { NVV }
\cs_new_protected:Npn \__pxtab_row_to_cells_seq:VN #1#2
{
	\seq_set_split:NVV #2 \l__pxtab_cell_sep_tl #1
}
\cs_new_protected:Npn \__pxtab_row_to_cells_seq:nN #1#2
{
	\tl_set:Nn \l__pxtab_tmpa_tl { #1 }
	\__pxtab_row_to_cells_seq:VN \l__pxtab_tmpa_tl #2
}
\prg_new_conditional:Npnn \__pxtab_row_has_content:n #1 { TF }
{
	\__pxtab_row_to_cells_seq:nN { #1 } \l__pxtab_cells_seq
	\bool_set_false:N \l__pxtab_simple_colspec_bool
	\seq_map_inline:Nn \l__pxtab_cells_seq
	{
		\tl_if_blank:nF { ##1 }
			{ \bool_set_true:N \l__pxtab_simple_colspec_bool }
	}
	\bool_if:NTF \l__pxtab_simple_colspec_bool
		{ \prg_return_true: }
		{ \prg_return_false: }
}
\cs_new_protected:Npn \__pxtab_trim_trailing_empty_rows:
{
	\seq_if_empty:NF \l__pxtab_rows_seq
		{
			\seq_get_right:NN \l__pxtab_rows_seq \l__pxtab_tmpa_tl
			\exp_args:NV \__pxtab_row_has_content:nTF \l__pxtab_tmpa_tl
				{ }
				{
					\seq_pop_right:NN \l__pxtab_rows_seq \l__pxtab_tmpa_tl
					\__pxtab_trim_trailing_empty_rows:
				}
		}
}
\cs_new_protected:Npn \__pxtab_parse_rows_from_list:n #1
{
	\seq_clear:N \l__pxtab_rows_seq
	\tl_set:Nn \l__pxtab_tmpa_tl { #1 }
	\seq_set_split:NVV \l__pxtab_rows_seq \l__pxtab_row_sep_tl \l__pxtab_tmpa_tl
	\__pxtab_trim_trailing_empty_rows:
}
\prg_new_conditional:Npnn \__pxtab_prepare_ncols: { TF }
{
	\seq_if_empty:NTF \l__pxtab_rows_seq
		{
			\__pxtab_warn_empty_table:
			\prg_return_false:
		}
		{
			\seq_get_left:NN \l__pxtab_rows_seq \l__pxtab_tmpa_tl
			\__pxtab_row_to_cells_seq:VN \l__pxtab_tmpa_tl \l__pxtab_cells_seq
			\int_set:Nn \l__pxtab_ncols_int { \seq_count:N \l__pxtab_cells_seq }
			\int_compare:nNnTF { \l__pxtab_ncols_int } = { 0 }
				{
					\__pxtab_warn_empty_row:
					\prg_return_false:
				}
				{ \prg_return_true: }
		}
}

\cs_new:Npn \__pxtab_emit_cell:nnn #1#2#3
{
	{
		#3
		\tl_if_blank:eTF { #2 }
			{ #1 }
			{ \textcolor{#2}{#1} }
	}
}
\cs_new_protected:Npn \__pxtab_append_row_to_body:nnnn #1#2#3#4
{
	\__pxtab_row_to_cells_seq:nN { #1 } \l__pxtab_cells_seq
	\tl_clear:N \l__pxtab_tmpa_tl
	\int_set:Nn \l__pxtab_tmpa_int { \seq_count:N \l__pxtab_cells_seq }
	\int_step_inline:nn { #2 }
	{
		\int_compare:nNnT { ##1 } > { 1 }
			{ \tl_put_right:Nx \l__pxtab_tmpa_tl { \__pxtab_alignment_tab: } }
		\int_compare:nNnTF { ##1 } > { \l__pxtab_tmpa_int }
			{ }
			{
				\tl_set:Nx \l__pxtab_cell_tmp_tl
					{ \seq_item:Nn \l__pxtab_cells_seq { ##1 } }
				\tl_put_right:Nx \l__pxtab_tmpa_tl
					{
						\__pxtab_emit_cell:nnn
							{ \exp_not:V \l__pxtab_cell_tmp_tl }
							{ \exp_not:n { #3 } }
							{ \exp_not:n { #4 } }
					}
			}
	}
	\tl_put_right:Nx \l__pxtab_body_tl { \exp_not:V \l__pxtab_tmpa_tl }
}
\cs_new_protected:Npn \__pxtab_append_row_styled_to_body:nnnnn #1#2#3#4#5
{
	\tl_if_blank:eF { #3 }
		{
			% Keep background attachment at row scope so color spans
			% the full row/cell area (not only individual cell content).
			\tl_put_right:Nx \l__pxtab_body_tl
				{
					\exp_not:N \rowcolor { #3 }
					\exp_not:N \ignorespaces
				}
		}
	\__pxtab_probe_event:nn { styled-row-bg } { #3 }
	\__pxtab_probe_event:nn { styled-row-text } { #4 }
	\__pxtab_probe_event:nn { styled-row-font } { #5 }
	\__pxtab_append_row_to_body:nnnn { #1 } { #2 } { #4 } { #5 }
}
\cs_new_protected:Npn \__pxtab_emit_row_direct:nnnn #1#2#3#4
{
	\__pxtab_row_to_cells_seq:nN { #1 } \l__pxtab_cells_seq
	\int_set:Nn \l__pxtab_tmpa_int { \seq_count:N \l__pxtab_cells_seq }
	\int_step_inline:nn { #2 }
	{
		\int_compare:nNnT { ##1 } > { 1 } { \__pxtab_alignment_tab: }
		\int_compare:nNnTF { ##1 } > { \l__pxtab_tmpa_int }
			{ }
			{
				\__pxtab_emit_cell:nnn
					{ \seq_item:Nn \l__pxtab_cells_seq { ##1 } }
					{ #3 }
					{ #4 }
			}
	}
}

\cs_new_protected:Npn \__pxtab_count_simple_colspec:
{
	\int_zero:N \l__pxtab_colspec_count_int
	\tl_set_eq:NN \l__pxtab_tmpa_tl \l__pxtab_colspec_tl
	\bool_set_true:N \l__pxtab_simple_colspec_bool
	\tl_map_inline:Nn \l__pxtab_tmpa_tl
	{
		\token_if_space:NTF ##1
			{ }
			{
				\str_case:nnF { ##1 }
				{
					{ c } { \int_incr:N \l__pxtab_colspec_count_int }
					{ l } { \int_incr:N \l__pxtab_colspec_count_int }
					{ r } { \int_incr:N \l__pxtab_colspec_count_int }
				}
				{ \bool_set_false:N \l__pxtab_simple_colspec_bool }
			}
	}
}
\prg_new_conditional:Npnn \__pxtab_width_is_natural: { TF }
{
	\tl_set_eq:NN \l__pxtab_tmpa_tl \l__pxtab_width_tl
	\tl_trim_spaces:N \l__pxtab_tmpa_tl
	\tl_if_eq:NnTF \l__pxtab_tmpa_tl { natural }
		{ \prg_return_true: }
		{ \prg_return_false: }
}
\cs_generate_variant:Nn \str_case:nnF { xn }
\cs_new_protected:Npn \__pxtab_prepare_paragraph_column_width:n #1
{
	\__pxtab_width_is_natural:TF
		{
			\dim_set:Nn \l__pxtab_paragraph_colwidth_dim
				{ \dim_eval:n { \linewidth / #1 } }
		}
		{
			\dim_set:Nn \l__pxtab_paragraph_colwidth_dim
				{ \dim_eval:n { \l__pxtab_width_tl / #1 } }
		}
	\str_case:xnF { \tl_to_str:N \l__pxtab_auto_align_tl }
		{
			{ l } { \tl_set:Nn \l__pxtab_paragraph_align_spec_tl { \raggedright\arraybackslash } }
			{ c } { \tl_set:Nn \l__pxtab_paragraph_align_spec_tl { \centering\arraybackslash } }
			{ r } { \tl_set:Nn \l__pxtab_paragraph_align_spec_tl { \raggedleft\arraybackslash } }
		}
		{ \tl_set:Nn \l__pxtab_paragraph_align_spec_tl { \centering\arraybackslash } }
}
\cs_new_protected:Npn \__pxtab_build_paragraph_colspec:n #1
{
	\__pxtab_prepare_paragraph_column_width:n { #1 }
	\tl_clear:N \l__pxtab_colspec_tl
	\int_step_inline:nn { #1 }
	{
		\tl_put_right:Nx \l__pxtab_colspec_tl
			{
				>{ \exp_not:V \l__pxtab_paragraph_align_spec_tl }
				\tl_use:N \l__pxtab_paragraph_coltype_tl
				{ \dim_use:N \l__pxtab_paragraph_colwidth_dim }
			}
	}
}
\cs_new_protected:Npn \__pxtab_build_paragraph_box_colspec:nN #1#2
{
	\__pxtab_prepare_paragraph_column_width:n { #1 }
	\tl_clear:N #2
	\int_step_inline:nn { #1 }
	{
		\tl_put_right:Nn #2 { | }
		\tl_put_right:Nx #2
			{
				>{ \exp_not:V \l__pxtab_paragraph_align_spec_tl }
				\tl_use:N \l__pxtab_paragraph_coltype_tl
				{ \dim_use:N \l__pxtab_paragraph_colwidth_dim }
			}
	}
	\tl_put_right:Nn #2 { | }
}
\cs_new_protected:Npn \__pxtab_append_auto_column_to_tl:N #1

	\str_case:xnF { \tl_to_str:N \l__pxtab_auto_align_tl }

			{ c } { \tl_put_right:Nn #1 { c } }
			{ r } { \tl_put_right:Nn #1 { r } }
		}
		{ \tl_put_right:Nn #1 { c } }
}
\cs_new_protected:Npn \__pxtab_build_auto_colspec:n #1
{
	\tl_clear:N \l__pxtab_colspec_tl
	\int_step_inline:nn { #1 }

}
\cs_new_protected:Npn \__pxtab_build_auto_box_colspec:nN #1#2
{
	\tl_clear:N #2
	\tl_put_right:Nn #2 { | }
	\int_step_inline:nn { #1 }
		{
			\__pxtab_append_auto_column_to_tl:N #2
			\tl_put_right:Nn #2 { | }
		}
}
\cs_new_protected:Npn \__pxtab_prepare_layout_mode:
{
	\bool_set_false:N \l__pxtab_use_tabularx_bool
	\__pxtab_width_is_natural:TF
		{ }
		{ \bool_set_true:N \l__pxtab_use_tabularx_bool }
}
\cs_new_protected:Npn \__pxtab_validate_or_build_colspec:n #1
{
	\bool_set_false:N \l__pxtab_auto_colspec_bool
	\tl_if_empty:NF \l__pxtab_colspec_tl
	{
		\__pxtab_count_simple_colspec:
		\bool_if:NT \l__pxtab_simple_colspec_bool
		{
			\int_compare:nNnF { \l__pxtab_colspec_count_int } = { #1 }
			{
				\msg_warning:nn { pxTAB } { colspec-mismatch }
				\tl_clear:N \l__pxtab_colspec_tl
			}
		}
	}
	\tl_if_empty:NT \l__pxtab_colspec_tl
	{
		\bool_set_true:N \l__pxtab_auto_colspec_bool
		\tl_if_eq:NnTF \l__pxtab_cellmode_tl { paragraph }
			{ \__pxtab_build_paragraph_colspec:n { #1 } }
			{ \__pxtab_build_auto_colspec:n { #1 } }
	}
}
\cs_new_protected:Npn \__pxtab_build_box_colspec:NN #1#2
{
	\tl_set:Nx \l__pxtab_tmpa_tl { \tl_use:N #2 }
	\bool_set_true:N \l__pxtab_simple_colspec_bool
	\tl_map_inline:Nn \l__pxtab_tmpa_tl
	{
		\token_if_space:NTF ##1
			{ }
			{
				\str_case:nnF { ##1 }
				{ { c } { } { l } { } { r } { } }
				{ \bool_set_false:N \l__pxtab_simple_colspec_bool }
			}
	}
	\bool_if:NTF \l__pxtab_simple_colspec_bool
	{
		\tl_clear:N #1
		\tl_put_right:Nn #1 { | }
		\tl_map_inline:Nn \l__pxtab_tmpa_tl
			{ \tl_put_right:Nn #1 { ##1| } }
	}
	{ \tl_set:Nx #1 { | \tl_use:N #2 | } }
}
