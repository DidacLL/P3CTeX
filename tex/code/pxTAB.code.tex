\keys_define:nn { pxTAB }
{
	style          .choices:nn = { plain, header, striped, boxed, grid }
		{ \tl_set:Nn \l__pxtab_style_tl { #1 } },
	style          .initial:n  = plain,
	headerbgcolor  .tl_set:N   = \l__pxtab_header_bg_tl,
	headerbgcolor  .initial:n  = { gray!25 },
	headertextcolor .tl_set:N  = \l__pxtab_header_text_tl,
	headertextcolor .initial:n = { black },
	altrowbgcolor  .tl_set:N   = \l__pxtab_alt_row_bg_tl,
	altrowbgcolor  .initial:n  = { gray!10 },
	altrowcolor    .meta:n     = { altrowbgcolor = {#1} },
	altrowtextcolor .tl_set:N  = \l__pxtab_alt_row_text_tl,
	altrowtextcolor .initial:n = {},
	colspec        .tl_set:N   = \l__pxtab_colspec_tl,
	colspec        .initial:n  = {},
	rowsep         .tl_set:N   = \l__pxtab_row_sep_tl,
	rowsep         .initial:n  = {;},
	cellsep        .tl_set:N   = \l__pxtab_cell_sep_tl,
	cellsep        .initial:n  = {,},
	unknown        .code:n     = { }
}

%% ===== PARSING HELPERS ============
%% =================================
\cs_generate_variant:Nn \seq_set_split:Nnn { NVV }
\cs_new_protected:Npn \__pxtab_row_to_cells_seq:VN #1#2
{
	\seq_set_split:NVV #2 \l__pxtab_cell_sep_tl #1
}
\cs_new_protected:Npn \__pxtab_row_to_cells_seq:nN #1#2
{
	\tl_set:Nn \l__pxtab_tmpa_tl { #1 }
	\__pxtab_row_to_cells_seq:VN \l__pxtab_tmpa_tl #2
}
\cs_new_protected:Npn \__pxtab_parse_rows_from_list:n #1
{
	\seq_clear:N \l__pxtab_rows_seq
	\tl_set:Nn \l__pxtab_tmpa_tl { #1 }
	\seq_set_split:NVV \l__pxtab_rows_seq \l__pxtab_row_sep_tl \l__pxtab_tmpa_tl
}
\prg_new_conditional:Npnn \__pxtab_prepare_ncols: { TF }
{
	\seq_if_empty:NTF \l__pxtab_rows_seq
		{
			\__pxtab_warn_empty_table:
			\prg_return_false:
		}
		{
			\seq_get_left:NN \l__pxtab_rows_seq \l__pxtab_tmpa_tl
			\__pxtab_row_to_cells_seq:VN \l__pxtab_tmpa_tl \l__pxtab_cells_seq
			\int_set:Nn \l__pxtab_ncols_int { \seq_count:N \l__pxtab_cells_seq }
			\int_compare:nNnTF { \l__pxtab_ncols_int } = { 0 }
				{
					\__pxtab_warn_empty_row:
					\prg_return_false:
				}
				{ \prg_return_true: }
		}
}

%% ===== ROW EMISSION ===============
%% =================================
\cs_new:Npn \__pxtab_emit_cell:nn #1#2
{
	\tl_if_blank:nTF { #2 }
		{ #1 }
		{ \textcolor{#2}{#1} }
}
\cs_new_protected:Npn \__pxtab_append_row_to_body:nnn #1#2#3
{
	\__pxtab_row_to_cells_seq:nN { #1 } \l__pxtab_cells_seq
	\tl_clear:N \l__pxtab_tmpa_tl
	\int_set:Nn \l__pxtab_tmpa_int { \seq_count:N \l__pxtab_cells_seq }
	\int_step_inline:nn { #2 }
	{
		\int_compare:nNnT { ##1 } > { 1 }
			{ \tl_put_right:Nx \l__pxtab_tmpa_tl { \__pxtab_alignment_tab: } }
		\int_compare:nNnTF { ##1 } > { \l__pxtab_tmpa_int }
			{ }
			{
				\tl_put_right:Nx \l__pxtab_tmpa_tl
					{
						\__pxtab_emit_cell:nn
							{ \seq_item:Nn \l__pxtab_cells_seq { ##1 } }
							{ #3 }
					}
			}
	}
	\tl_put_right:Nx \l__pxtab_body_tl { \tl_use:N \l__pxtab_tmpa_tl }
	\tl_put_right:Nn \l__pxtab_body_tl { \\ }
}
\cs_new_protected:Npn \__pxtab_append_row_styled_to_body:nnnn #1#2#3#4
{
	\tl_if_blank:nF { #3 } { \tl_put_right:Nx \l__pxtab_body_tl { \exp_not:N \rowcolor { #3 } } }
	\__pxtab_append_row_to_body:nnn { #1 } { #2 } { #4 }
}
%% =================================
%% ===== COLUMN SPEC HELPERS =======
%% =================================
\cs_new_protected:Npn \__pxtab_count_simple_colspec:
{
	\int_zero:N \l__pxtab_colspec_count_int
	\tl_set:Nx \l__pxtab_tmpa_tl { \tl_remove_all_spaces:n { \l__pxtab_colspec_tl } }
	\bool_set_true:N \l__pxtab_simple_colspec_bool
	\tl_map_inline:Nn \l__pxtab_tmpa_tl
	{
		\str_case:nnF { ##1 }
		{
			{ c } { \int_incr:N \l__pxtab_colspec_count_int }
			{ l } { \int_incr:N \l__pxtab_colspec_count_int }
			{ r } { \int_incr:N \l__pxtab_colspec_count_int }
		}
		{ \bool_set_false:N \l__pxtab_simple_colspec_bool }
	}
}
\cs_new_protected:Npn \__pxtab_validate_or_build_colspec:n #1
{
	\tl_if_empty:NF \l__pxtab_colspec_tl
	{
		\__pxtab_count_simple_colspec:
		\bool_if:NT \l__pxtab_simple_colspec_bool
		{
			\int_compare:nNnF { \l__pxtab_colspec_count_int } = { #1 }
			{
				\msg_warning:nn { pxTAB } { colspec-mismatch }
				\tl_clear:N \l__pxtab_colspec_tl
			}
		}
	}
	\tl_if_empty:NT \l__pxtab_colspec_tl
		{ \tl_set:Nx \l__pxtab_colspec_tl { \prg_replicate:nn { #1 } { c } } }
}
\cs_new_protected:Npn \__pxtab_build_box_colspec:NN #1#2
{
	\tl_set:Nx \l__pxtab_tmpa_tl { \tl_remove_all_spaces:n { \tl_use:N #2 } }
	\bool_set_true:N \l__pxtab_simple_colspec_bool
	\tl_map_inline:Nn \l__pxtab_tmpa_tl
	{
		\str_case:nnF { ##1 }
		{ { c } { } { l } { } { r } { } }
		{ \bool_set_false:N \l__pxtab_simple_colspec_bool }
	}
	\bool_if:NTF \l__pxtab_simple_colspec_bool
	{
		\tl_clear:N #1
		\tl_put_right:Nn #1 { | }
		\tl_map_inline:Nn \l__pxtab_tmpa_tl
			{ \tl_put_right:Nn #1 { ##1| } }
	}
	{ \tl_set:Nx #1 { | \tl_use:N #2 | } }
}
\cs_new_protected:Npn \__pxtab_begin_tabular_from_tl:N #1
{
	\use:x { \exp_not:N \begin{tabular}{ \tl_use:N #1 } }
}
\cs_new_protected:Npn \__pxtab_end_tabular:
	{ \end{tabular} }
\cs_new_protected:Npn \__pxtab_emit_tabular_with_body:N #1
{
	\__pxtab_begin_tabular_from_tl:N #1
	\tl_use:N \l__pxtab_body_tl
	\__pxtab_end_tabular:
}

%% ===== STYLE RENDERERS ============

\cs_new_protected:Npn \__pxtab_render_table_plain:n #1
{
	\tl_clear:N \l__pxtab_body_tl
	\seq_map_inline:Nn \l__pxtab_rows_seq
		{ \__pxtab_append_row_to_body:nnn { ##1 } { #1 } { } }
	\__pxtab_emit_tabular_with_body:N \l__pxtab_colspec_tl
}
\cs_new_protected:Npn \__pxtab_render_table_header:n #1
{
	\tl_clear:N \l__pxtab_body_tl
	\int_zero:N \l__pxtab_row_index_int
	\seq_map_inline:Nn \l__pxtab_rows_seq
	{
		\int_incr:N \l__pxtab_row_index_int
		\int_compare:nNnTF { \l__pxtab_row_index_int } = { 1 }
			{
				\__pxtab_append_row_styled_to_body:nnnn
					{ ##1 } { #1 }
					{ \l__pxtab_header_bg_tl }
					{ \l__pxtab_header_text_tl }
			}
			{ \__pxtab_append_row_to_body:nnn { ##1 } { #1 } { } }
	}
	\__pxtab_emit_tabular_with_body:N \l__pxtab_colspec_tl
}
\cs_new_protected:Npn \__pxtab_render_table_striped:n #1
{
	\tl_clear:N \l__pxtab_body_tl
	\int_zero:N \l__pxtab_row_index_int
	\seq_map_inline:Nn \l__pxtab_rows_seq
	{
		\int_incr:N \l__pxtab_row_index_int
		\int_if_odd:nTF { \l__pxtab_row_index_int }
			{
				\__pxtab_append_row_styled_to_body:nnnn
					{ ##1 } { #1 }
					{ \l__pxtab_alt_row_bg_tl }
					{ \l__pxtab_alt_row_text_tl }
			}
			{ \__pxtab_append_row_to_body:nnn { ##1 } { #1 } { } }
	}
	\__pxtab_emit_tabular_with_body:N \l__pxtab_colspec_tl
}
\cs_new_protected:Npn \__pxtab_render_table_boxed:n #1
{
	\tl_clear:N \l__pxtab_body_tl
	\__pxtab_build_box_colspec:NN \l__pxtab_tmpb_tl \l__pxtab_colspec_tl
	\seq_map_inline:Nn \l__pxtab_rows_seq
		{ \__pxtab_append_row_to_body:nnn { ##1 } { #1 } { } }
	\tl_put_left:Nn \l__pxtab_body_tl { \hline }
	\tl_put_right:Nn \l__pxtab_body_tl { \hline }
	\__pxtab_emit_tabular_with_body:N \l__pxtab_tmpb_tl
}
\cs_new_protected:Npn \__pxtab_render_table_grid:n #1
{
	\tl_clear:N \l__pxtab_body_tl
	\__pxtab_build_box_colspec:NN \l__pxtab_tmpb_tl \l__pxtab_colspec_tl
	\tl_put_right:Nn \l__pxtab_body_tl { \hline }
	\seq_map_inline:Nn \l__pxtab_rows_seq
	{
		\__pxtab_append_row_to_body:nnn { ##1 } { #1 } { }
		\tl_put_right:Nn \l__pxtab_body_tl { \hline }
	}
	\__pxtab_emit_tabular_with_body:N \l__pxtab_tmpb_tl
}

\cs_new_protected:Npn \__pxtab_table_from_seq:n #1
{
	\__pxtab_validate_or_build_colspec:n { #1 }
	\str_set:Nx \l__pxtab_style_str { \tl_use:N \l__pxtab_style_tl }
	\str_case:nnF { \str_use:N \l__pxtab_style_str }
	{
		{ plain }   { \__pxtab_render_table_plain:n { #1 } }
		{ header }  { \__pxtab_render_table_header:n { #1 } }
		{ striped } { \__pxtab_render_table_striped:n { #1 } }
		{ boxed }   { \__pxtab_render_table_boxed:n { #1 } }
		{ grid }    { \__pxtab_render_table_grid:n { #1 } }
	}
	{ \__pxtab_render_table_plain:n { #1 } }
}

\cs_new_protected:Npn \pxTAB_table_matrix:nnn #1#2#3
{
	\int_compare:nNnTF { #1 } > { 0 }
	{
		\int_compare:nNnTF { #2 } > { 0 }
		{
			\tl_set:Nn \l__pxtab_tmpa_tl { #3 }
			\seq_set_split:NVV \l__pxtab_cells_seq \l__pxtab_cell_sep_tl \l__pxtab_tmpa_tl
			\seq_clear:N \l__pxtab_rows_seq
			\int_step_inline:nn { #1 }
			{
				\tl_clear:N \l__pxtab_tmpa_tl
				\int_step_inline:nn { #2 }
				{
					\int_compare:nNnT { ####1 } > { 1 }
						{ \tl_put_right:NV \l__pxtab_tmpa_tl \l__pxtab_cell_sep_tl }
					\int_set:Nn \l__pxtab_tmpa_int
						{ ( ##1 - 1 ) * #2 + ####1 }
					\int_compare:nNnTF
						{ \l__pxtab_tmpa_int } > { \seq_count:N \l__pxtab_cells_seq }
						{ }
						{
							\tl_put_right:Nx \l__pxtab_tmpa_tl
								{ \seq_item:Nn \l__pxtab_cells_seq { \l__pxtab_tmpa_int } }
						}
				}
				\seq_put_right:NV \l__pxtab_rows_seq \l__pxtab_tmpa_tl
			}
			\__pxtab_table_from_seq:n { #2 }
		}
		{ \__pxtab_warn_empty_row: }
	}
	{ \__pxtab_warn_empty_table: }
}
