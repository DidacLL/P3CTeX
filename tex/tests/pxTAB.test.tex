\section*{pxTAB redesign regression}

\subsection*{1. Plain style}
\pxTABsetup{style=plain}
\pxTable{A, B, C}{1, 2, 3}{4, 5, 6}

\subsection*{2. Header style (background + text color)}
\pxTABsetup{style=header,headerbgcolor=blue!20,headertextcolor=blue!70!black}
\pxTable{Name, Age, City}{Alice, 20, Barcelona}{Bob, 22, Madrid}

\subsection*{3. Striped style}
\pxTABsetup{style=striped,altrowbgcolor=gray!15}
\pxTable{R1C1, R1C2}{R2C1, R2C2}{R3C1, R3C2}{R4C1, R4C2}

\subsection*{4. Boxed style}
\pxTABsetup{style=boxed}
\pxTable{H1, H2, H3}{a, b, c}{d, e, f}

\subsection*{5. Grid style}
\pxTABsetup{style=grid}
\pxTable{G1, G2}{x, y}{u, v}

\subsection*{6. From list}
\pxTABsetup{style=plain}
\pxTableFromList{ID, Name; 1, Alice; 2, Bob}

\subsection*{7. From list with custom separators}
\pxTABsetup{style=plain,rowsep=|,cellsep=/}
\pxTableFromList{K1/K2|A/B|C/D}

\subsection*{8. Single row and single column}
\pxTABsetup{style=plain,rowsep={;},cellsep={,}}
\pxTableRow{Only, one, row}
\pxTable{One}{Two}{Three}

\subsection*{9. Row length mismatch (pad/truncate)}
\pxTable{A, B, C}{1, 2}{x, y, z, overflow}

\subsection*{10. Cell with comma}
\pxTable{{Smith, John}, Age}{20, Student}

\subsection*{11. Head/body commands}
\pxTABsetup{style=header,headerbgcolor=red!15,headertextcolor=red!60!black}
\pxTableHeadBody{Name, Role}{Alice, Dev; Bob, QA}
\pxTableHeadBodyRows{Name, Score}{Anna, 8}{Marc, 9}{Joan, 10}

\subsection*{12. Matrix command}
\pxTABsetup{style=grid}
\pxTableMatrix{2}{3}{m11,m12,m13,m21,m22,m23}

\subsection*{13. Local options do not leak globally}
\pxTABsetup{style=plain}
\pxTable[style=header,headerbgcolor=green!20,headertextcolor=green!50!black]{L1, L2}{v1, v2}
\pxTable{P1, P2}{x1, x2}

\subsection*{14. Matrix sizing with exact and short cell lists}
\pxTABsetup{style=plain}
\pxTableMatrix{2}{2}{n11,n12,n21,n22}
\pxTableMatrix{2}{3}{s11,s12,s13,s21}

\subsection*{15. Striped text color and alias key}
\pxTABsetup{style=striped,altrowcolor=yellow!20,altrowtextcolor=purple}
\pxTable{S1, S2}{aa, bb}{cc, dd}{ee, ff}

\subsection*{16. Command-level options remain supported}
\pxTableFromList[style=grid]{C1, C2; 1, 2; 3, 4}
\pxTableHeadBodyRows[style=boxed]{H1, H2}{b1, b2}{b3, b4}
\pxTableMatrix[style=plain,cellsep=/]{2}{2}{x1/x2/x3/x4}

\subsection*{17. Unknown key is ignored}
\pxTABsetup{style=plain,unknownkey=ignored}
\pxTable{U1, U2}{ok, still works}

\subsection*{18. Width control with auto columns}
\pxTABsetup{style=plain,width=\linewidth,align=l}
\pxTable{Long header A, Long header B, Long header C}
        {This text should wrap in the first column,Data,123}
        {Another long text cell to check width behavior,More data,456}

\subsection*{19. Local font size + row stretch}
\pxTable[fontsize=\small,rowstretch=1.05,style=header]
        {Name, Notes}
        {Alice, Compact rows with smaller text}
        {Bob, This checks per-table sizing controls}

\subsection*{20. Explicit colspec override with width}
\pxTABsetup{style=plain,width=\linewidth,colspec=ll}
\pxTable{Col 1, Col 2}{Width key remains active while explicit colspec is preserved}
