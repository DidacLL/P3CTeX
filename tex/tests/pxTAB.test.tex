\documentclass{article}

\usepackage{pxTAB}

\begin{document}

\section*{pxTAB redesign regression}

\subsection*{1. Plain style}
\pxTABsetup{style=plain}
\pxTable{A, B, C}{1, 2, 3}{4, 5, 6}

\subsection*{2. Header style (background + text color)}
\pxTABsetup{style=header,headerbgcolor=blue!20,headertextcolor=blue!70!black}
\pxTable{Name, Age, City}{Alice, 20, Barcelona}{Bob, 22, Madrid}

\subsection*{3. Striped style}
\pxTABsetup{style=striped,altrowbgcolor=gray!15}
\pxTable{R1C1, R1C2}{R2C1, R2C2}{R3C1, R3C2}{R4C1, R4C2}

\subsection*{4. Boxed style}
\pxTABsetup{style=boxed}
\pxTable{H1, H2, H3}{a, b, c}{d, e, f}

\subsection*{5. Grid style}
\pxTABsetup{style=grid}
\pxTable{G1, G2}{x, y}{u, v}

\subsection*{6. From list}
\pxTABsetup{style=plain}
\pxTableFromList{ID, Name; 1, Alice; 2, Bob}

\subsection*{7. From list with custom separators}
\pxTABsetup{style=plain,rowsep=|,cellsep=/}
\pxTableFromList{K1/K2|A/B|C/D}

\subsection*{8. Single row and single column}
\pxTABsetup{style=plain,rowsep={;},cellsep={,}}
\pxTableRow{Only, one, row}
\pxTable{One}{Two}{Three}

\subsection*{9. Row length mismatch (pad/truncate)}
\pxTable{A, B, C}{1, 2}{x, y, z, overflow}

\subsection*{10. Cell with comma}
\pxTable{{Smith, John}, Age}{20, Student}

\subsection*{11. Head/body commands}
\pxTABsetup{style=header,headerbgcolor=red!15,headertextcolor=red!60!black}
\pxTableHeadBody{Name, Role}{Alice, Dev; Bob, QA}
\pxTableHeadBodyRows{Name, Score}{Anna, 8}{Marc, 9}{Joan, 10}

\subsection*{12. Matrix command}
\pxTABsetup{style=grid}
\pxTableMatrix{2}{3}{m11,m12,m13,m21,m22,m23}

\end{document}
