\documentclass[PR,data=fake,UML]{P3CTeX}
%\PECTeXconfig{cover-config={fullcover}}
%\documentclass{article}
%\usepackage[UML]{pxCORE}
\begin{document}
\pxUMLClass{Person}[Animal]{- name : String}[+ getName() : String]  % STANDARD CLASS
\pxUMLClass[abstractclass]{Animal}{- name : String}[+ getName() : String]  % ABSTRACT CLASS
\pxUMLClass[enumclass]{EyeColor}{Brown, Blue, Green} %ENUM
\pxUMLClass[interface]{Science}{- id : int}[+ solveProblem(): int] %INTERFACE
\pxUMLClass[interface]{Knowledge}{- extra : String}
\pxUMLClass{KnowledgeImpl}{}[+ solveSpecificProblem(): int]<Knowledge> %IMPLEMENTATION
%%   ADD NEW PROPERTIES
\addAttribute{Person}{- age : Integer}
\addInheritance{Knowledge}{Science}
%%   DRAW DIAGRAM
\pxUMLDiagram{

	\drawsimpleclass*{Animal}[-6,3]
	\drawclass[text width=12em]{Animal}[0,0]
	\drawclass[text width=12em]{Person}[0,-5]
	\drawclass[text width=10em]{EyeColor}[-6,-3]

	\drawclass*{Science}[5,5]
	\drawsimpleclass{Knowledge}[10,0]
	\drawclass{KnowledgeImpl}[10,-5]
	%%  RELATIONS
	\association{Animal}{0..*}{is known }{Knowledge}{knows}{1..*}
	\composition{Person}{1}{*}{EyeColor}
	\aggregation{Person}{Learned}{1..*}{KnowledgeImpl}

}[.9\linewidth]<-4em>

\end{document}
