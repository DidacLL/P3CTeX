  {
    \iow_term:x
      {
        PXTAB_PROBE:SCENARIO=\tl_use:N \g__pxtab_probe_scenario_tl ;
        STYLE=\tl_use:N \l__pxtab_style_tl ;

        RULEWIDTH=\dim_use:N \l__pxtab_rule_width_dim ;
        FONTSIZE=\tl_to_str:N \l__pxtab_font_size_tl ;

        BODY=\tl_to_str:N \l__pxtab_body_tl
      }

  }
