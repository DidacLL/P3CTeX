\bool_new:N \g__pxtab_probe_blank_row_seen_bool
\cs_new_protected:Npn \__pxtab_probe_pass:n #1
  { \iow_term:x { PXTAB_ASSERT~PASS:~#1 } }
\cs_new_protected:Npn \__pxtab_probe_fail:n #1
  {
    \iow_term:x { PXTAB_ASSERT~FAIL:~#1 }
    \PackageError { pxTAB-style-probe } { #1 } { Check~pxTAB~row/style~token~emission. }
  }
\cs_new_protected:Npn \__pxtab_probe_assert_eq:nnn #1#2#3
  {
    \int_compare:nNnTF { #1 } = { #2 }
      { \__pxtab_probe_pass:n { #3 } }
      { \__pxtab_probe_fail:n { #3 } }
  }
\cs_new_protected:Npn \__pxtab_probe_assert_false:Nn #1#2
  {
    \bool_if:NTF #1
      { \__pxtab_probe_fail:n { #2 } }
      { \__pxtab_probe_pass:n { #2 } }
  }
\cs_new_protected:Npn \__pxtab_probe_assert_body_not_contains:n #1
  {
    \tl_if_in:NnTF \l__pxtab_body_tl { #1 }
      { \__pxtab_probe_fail:n { body-contains-forbidden-token-\tl_to_str:n {#1} } }
      { \__pxtab_probe_pass:n { body-token-absent-\tl_to_str:n {#1} } }
  }
\cs_new_protected:Npn \__pxtab_probe_assert_body_contains:n #1
  {
    \tl_if_in:NnTF \l__pxtab_body_tl { #1 }
      { \__pxtab_probe_pass:n { body-token-present-\tl_to_str:n {#1} } }
      { \__pxtab_probe_fail:n { body-missing-required-token-\tl_to_str:n {#1} } }
  }
\cs_new_protected:Npn \pxProbeScenario #1#2
  {
    \tl_gset:Nn \g__pxtab_probe_scenario_tl { #1 }
    \int_gset:Nn \g__pxtab_probe_expected_rows_int { #2 }
    \int_gzero:N \g__pxtab_probe_seen_rows_int
    \bool_gset_false:N \g__pxtab_probe_blank_row_seen_bool
    \iow_term:x { PXTAB_PROBE:SCENARIO-BEGIN=#1 }
  }
\cs_set_protected:Npn \__pxtab_probe_event:nn #1#2
  { \iow_term:x { PXTAB_EVENT:#1 } }

\cs_new_eq:NN \__pxtab_probe_append_row_orig:nnnn \__pxtab_append_row_to_body:nnnn
\cs_set_protected:Npn \__pxtab_append_row_to_body:nnnn #1#2#3#4
  {
    \int_gincr:N \g__pxtab_probe_seen_rows_int
    \tl_if_blank:nT { #1 } { \bool_gset_true:N \g__pxtab_probe_blank_row_seen_bool }
    \__pxtab_probe_append_row_orig:nnnn { #1 } { #2 } { #3 } { #4 }
  }

\cs_new_eq:NN \__pxtab_probe_emit_orig:N \__pxtab_emit_tabular_with_body:N
\cs_set_protected:Npn \__pxtab_emit_tabular_with_body:N #1
  {
    \__pxtab_probe_assert_eq:nnn
      { \g__pxtab_probe_seen_rows_int }
      { \g__pxtab_probe_expected_rows_int }
      { no-trailing-empty-row }
    \__pxtab_probe_assert_false:Nn
      \g__pxtab_probe_blank_row_seen_bool
      { no-blank-row-emitted }
    \tl_if_eq:NnT \l__pxtab_style_tl { boxed }
      {
        \__pxtab_probe_assert_eq:nnn
          { \g__pxtab_probe_seen_rows_int }
          { \g__pxtab_probe_expected_rows_int }
          { boxed-terminal-rowcount-clean }
        \__pxtab_probe_assert_body_not_contains:n { \crcr }
      }
    \tl_if_eq:NnT \l__pxtab_style_tl { grid }
      {
        \__pxtab_probe_assert_eq:nnn
          { \g__pxtab_probe_seen_rows_int }
          { \g__pxtab_probe_expected_rows_int }
          { grid-terminal-rowcount-clean }
        \__pxtab_probe_assert_body_not_contains:n { \crcr }
      }
    \tl_if_eq:NnT \l__pxtab_style_tl { header-boxed }
      {
        \__pxtab_probe_assert_eq:nnn
          { \g__pxtab_probe_seen_rows_int }
          { \g__pxtab_probe_expected_rows_int }
          { header-boxed-terminal-rowcount-clean }
        \__pxtab_probe_assert_body_not_contains:n { \crcr }
      }
    \tl_if_eq:NnT \l__pxtab_style_tl { header-grid }
      {
        \__pxtab_probe_assert_eq:nnn
          { \g__pxtab_probe_seen_rows_int }
          { \g__pxtab_probe_expected_rows_int }
          { header-grid-terminal-rowcount-clean }
        \__pxtab_probe_assert_body_not_contains:n { \crcr }
      }
    \tl_if_eq:NnT \l__pxtab_style_tl { striped }
      {
        \__pxtab_probe_assert_body_contains:n { \rowcolor }
        \__pxtab_probe_assert_body_not_contains:n { \cellcolor }
        \__pxtab_probe_assert_body_not_contains:n { \crcr }
      }
    \tl_if_eq:NnT \l__pxtab_style_tl { header-striped }
      {
        \__pxtab_probe_assert_body_contains:n { \rowcolor }
        \__pxtab_probe_assert_body_not_contains:n { \cellcolor }
        \__pxtab_probe_assert_body_not_contains:n { \crcr }
      }
    \iow_term:x
      {
        PXTAB_PROBE:SCENARIO=\tl_use:N \g__pxtab_probe_scenario_tl ;
        STYLE=\tl_use:N \l__pxtab_style_tl ;
        RULECOLOR=\tl_use:N \l__pxtab_rule_color_tl ;
        RULEWIDTH=\dim_use:N \l__pxtab_rule_width_dim ;
        FONTSIZE=\tl_to_str:N \l__pxtab_font_size_tl ;
        ARRAYSTRETCH=\tl_use:N \l__pxtab_arraystretch_tl ;
        COLSPEC=\tl_to_str:n { \tl_use:N #1 } ;
        BODY=\tl_to_str:N \l__pxtab_body_tl
      }
    \__pxtab_probe_emit_orig:N #1
  }
