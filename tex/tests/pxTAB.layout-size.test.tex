  {
    \iow_term:x { PXTAB-CHECK~FAIL:~#1 }

  }

  {
    \dim_compare:nNnTF { #1 } > { #2 }
      { \__pxtab_test_pass:n { #3 } }
      { \__pxtab_test_fail:n { #3 } }
  }

  {
    \dim_compare:nNnTF { \dim_abs:n { #1 - #2 } } < { 1pt }
      { \__pxtab_test_pass:n { #3 } }
      { \__pxtab_test_fail:n { #3 } }
  }

  }
