\section*{pxLST inline code test}

\subsection*{1. Single snippet (default style/language)}

\subsection*{2. With style override}
\pxCodeInline[style=light]{echo hello} and \pxCodeInline[style=plain]{void main}.

\pxCodeInline[language=pxBash]{echo "done"} and \pxCodeInline[language=pxJava]{String s = "test";}.

\subsection*{4. With preset}
\pxCodeInline[preset=minimal]{int x = 0;} (minimal = plain + pxJava).

\subsection*{5. Empty options}
\pxCodeInline[]{int n = 42;} (uses global \pxLSTsetup).
