\ExplSyntaxOn

%%% ---- Assertion infrastructure ----

\cs_new_protected:Npn \__pxtab_preset_test_pass:n #1
  { \iow_term:x { PXTAB_ASSERT~PASS:~#1 } }

\cs_new_protected:Npn \__pxtab_preset_test_fail:n #1
  {
    \iow_term:x { PXTAB_ASSERT~FAIL:~#1 }
    \PackageError { pxTAB-preset-test } { #1 }
      { Preset~test~assertion~failed. }
  }

\cs_new_protected:Npn \__pxtab_preset_test_assert_tl_eq:Nnn #1#2#3
  {
    \tl_if_eq:NnTF #1 { #2 }
      { \__pxtab_preset_test_pass:n { #3 } }
      {
        \iow_term:x
          { PXTAB_ASSERT~FAIL:~#3~(got=\tl_to_str:N #1) }
        \PackageError { pxTAB-preset-test } { #3 }
          { Preset~test~assertion~failed. }
      }
  }

\cs_new_protected:Npn \__pxtab_preset_test_assert_bool_true:Nn #1#2
  {
    \bool_if:NTF #1
      { \__pxtab_preset_test_pass:n { #2 } }
      { \__pxtab_preset_test_fail:n { #2 } }
  }

\cs_new_protected:Npn \__pxtab_preset_test_assert_bool_false:Nn #1#2
  {
    \bool_if:NTF #1
      { \__pxtab_preset_test_fail:n { #2 } }
      { \__pxtab_preset_test_pass:n { #2 } }
  }

\cs_new_protected:Npn \__pxtab_preset_test_assert_dim_eq:nnn #1#2#3
  {
    \dim_compare:nNnTF { \dim_abs:n { #1 - #2 } } < { 0.1pt }
      { \__pxtab_preset_test_pass:n { #3 } }
      { \__pxtab_preset_test_fail:n { #3 } }
  }

%%% ---- Capture state ----

\tl_new:N   \g__pxtab_preset_test_style_tl
\tl_new:N   \g__pxtab_preset_test_header_bg_tl
\tl_new:N   \g__pxtab_preset_test_header_text_tl
\tl_new:N   \g__pxtab_preset_test_alt_row_bg_tl
\tl_new:N   \g__pxtab_preset_test_arraystretch_tl
\dim_new:N  \g__pxtab_preset_test_tabcolsep_dim
\bool_new:N \g__pxtab_preset_test_center_bool
\bool_new:N \g__pxtab_preset_test_capture_bool

%%% ---- Probe hook: intercept emit to snapshot runtime key state ----

\cs_new_eq:NN \__pxtab_preset_test_emit_orig:N
  \__pxtab_emit_tabular_with_body:N

\cs_set_protected:Npn \__pxtab_emit_tabular_with_body:N #1
  {
    \bool_if:NT \g__pxtab_preset_test_capture_bool
      {
        \tl_gset_eq:NN  \g__pxtab_preset_test_style_tl
          \l__pxtab_style_tl
        \tl_gset_eq:NN  \g__pxtab_preset_test_header_bg_tl
          \l__pxtab_header_bg_tl
        \tl_gset_eq:NN  \g__pxtab_preset_test_header_text_tl
          \l__pxtab_header_text_tl
        \tl_gset_eq:NN  \g__pxtab_preset_test_alt_row_bg_tl
          \l__pxtab_alt_row_bg_tl
        \tl_gset_eq:NN  \g__pxtab_preset_test_arraystretch_tl
          \l__pxtab_arraystretch_tl
        \dim_gset:Nn \g__pxtab_preset_test_tabcolsep_dim
          { \l__pxtab_tabcolsep_dim }
        \bool_if:NTF \l__pxtab_center_bool
          { \bool_gset_true:N  \g__pxtab_preset_test_center_bool }
          { \bool_gset_false:N \g__pxtab_preset_test_center_bool }
      }
    \__pxtab_preset_test_emit_orig:N #1
  }

\ExplSyntaxOff

%% ====================================================================
%% 1. Save/use roundtrip
%%    Save a custom preset, apply via preset= key, verify probe state.
%% ====================================================================
\subsection*{1. Save/use roundtrip}

\pxTABsavepreset{custom-test}{%
  style=header,
  headerbgcolor=red!20,
  headertextcolor=red!70!black,
  arraystretch=1.5}

\ExplSyntaxOn
\bool_gset_true:N \g__pxtab_preset_test_capture_bool
\ExplSyntaxOff

\pxTable[preset=custom-test]{Name, Age}{Alice, 30}{Bob, 25}

\ExplSyntaxOn
\bool_gset_false:N \g__pxtab_preset_test_capture_bool
\__pxtab_preset_test_assert_tl_eq:Nnn
  \g__pxtab_preset_test_style_tl { header }
  { roundtrip-style-is-header }
\__pxtab_preset_test_assert_tl_eq:Nnn
  \g__pxtab_preset_test_header_bg_tl { red!20 }
  { roundtrip-headerbgcolor-is-red20 }
\__pxtab_preset_test_assert_tl_eq:Nnn
  \g__pxtab_preset_test_header_text_tl { red!70!black }
  { roundtrip-headertextcolor-is-red70black }
\__pxtab_preset_test_assert_tl_eq:Nnn
  \g__pxtab_preset_test_arraystretch_tl { 1.5 }
  { roundtrip-arraystretch-is-1.5 }
\ExplSyntaxOff
